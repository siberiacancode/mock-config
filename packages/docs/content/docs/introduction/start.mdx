## Installation

Install the Core package using your preferred package manager:

```package-install
mock-config-server
```

## Configuration

Create a `mock-server.config.ts` file with server configuration and export it as default.

```ts
import { FlatMockServerConfig } from 'mock-config-server';

const flatMockSeverConfig: FlatMockServerConfig = [
  {
    baseUrl: '/api'
  },
  {
    configs: [
      {
        path: '/user',
        method: 'get',
        routes: [{ data: { emoji: 'ğŸ¦', name: 'Nursultan' } }]
      }
    ]
  }
];

export default flatMockSeverConfig;
```
